Map<String, dynamic> toJson() {
  final map = <String, dynamic>{};
  map['id'] = id;
  map['name'] = name;
  map['description'] = description;
  map['metadata'] = metadata;
  map['secret'] = secret;
  map['pin'] = pin;
  map['is_private'] = isPrivate;
  map['viewers'] = viewers;
  map['enabled'] = enabled;
  map['type'] = type;
  if (media != null) {
    map['media'] = media?.map((v) => v.toJson()).toList();
  }
  return map;
}
