Map<String, dynamic> toJson() {
  final map = <String, dynamic>{};
  map['videoroom'] = videoroom;
  map['room'] = room;
  if (publishers != null) {
    map['publishers'] = publishers?.map((v) => v.toJson()).toList();
  }
  return map;
}
